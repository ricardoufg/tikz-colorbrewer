\documentclass{article}
\usepackage{tikz}
\usetikzlibrary{colorbrewer}
\usepackage{datatool}

\usepackage[active,tightpage]{preview}  %generates a tightly fitting border around the work
\PreviewEnvironment{tikzpicture}
\DTLloaddb[noheader=false]{scheme}{scheme_colors.txt}
\begin{document}
\begin{tikzpicture}
    \newcounter{row}
    \newcounter{col}
    \def\prevscheme{null}
    \DTLforeach*{scheme}{\scheme=Scheme,\colname=ColorName,\co=ColorLetter}
    {
      \ifthenelse%
        {\DTLiseq{\scheme}{\prevscheme}}%
        {}%
        { 
          \addtocounter{row}{1}
          \node[anchor=east] at (0,-\arabic{row}) {\scheme}; 
          \setcounter{col}{0}
        }

      \addtocounter{col}{1}
      \node[fill=\colname-\co, minimum width=1cm, minimum height=5mm] at
      (\arabic{col},-\arabic{row}) {}; 

      \let\prevscheme\scheme
    }
\end{tikzpicture}
\end{document}
